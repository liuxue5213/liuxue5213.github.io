function getSelectIds() {
var ids = [];
var rows = $(“#list”).datagrid(‘getSelections’);
for (var i = 0; i < rows.length; i++) {
ids.push(rows[i].id);
}
return ids.join(‘,’);
}

function compute(colName) {
var rows = $(‘#list’).datagrid(‘getRows’);
var total = 0;
for (var i = 0; i < rows.length; i++) {
total += parseFloat(rows[i][colName]);
}
return total;
}
